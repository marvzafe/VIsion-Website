// ======================================== // DETAIL PAGES - ADVANCED INTERACTIONS // ======================================== // This file adds extra magic to product-details.html and project-details.html document.addEventListener('DOMContentLoaded', () => { // ========== IMAGE SLIDER ENHANCEMENTS ========== const sliderContainer = document.querySelector('.slider-container'); const sliderWrapper = document.getElementById('slider-wrapper'); if (sliderWrapper) { // Add click-to-expand functionality const images = sliderWrapper.querySelectorAll('img'); images.forEach((img, index) => { img.addEventListener('click', () => { openImageModal(img.src); }); img.style.cursor = 'pointer'; }); // Keyboard navigation document.addEventListener('keydown', (e) => { if (e.key === 'ArrowLeft') moveSlide(-1); if (e.key === 'ArrowRight') moveSlide(1); }); // Add slide counter addSlideCounter(); } // ========== FEATURES LIST ANIMATION ========== const featuresList = document.getElementById('prod-features'); if (featuresList) { const items = featuresList.querySelectorAll('li'); items.forEach((item, index) => { item.style.animation = `slideFromLeft 0.6s ease-out ${index * 0.1}s both`; // Add hover effect item.addEventListener('mouseenter', function() { this.style.transform = 'translateX(10px)'; this.style.color = '#FF1817'; this.style.fontWeight = 'bold'; }); item.addEventListener('mouseleave', function() { this.style.transform = 'translateX(0)'; this.style.color = 'inherit'; this.style.fontWeight = 'normal'; }); }); } // ========== SPECS TABLE ANIMATIONS ========== const specTables = document.querySelectorAll('table'); specTables.forEach(table => { const rows = table.querySelectorAll('tbody tr'); rows.forEach((row, index) => { row.style.animation = `slideFromRight 0.7s ease-out ${index * 0.08}s both`; row.addEventListener('mouseenter', function() { this.style.background = 'rgba(255, 24, 23, 0.1)'; this.style.transform = 'translateX(5px)'; }); row.addEventListener('mouseleave', function() { this.style.background = 'transparent'; this.style.transform = 'translateX(0)'; }); }); }); // ========== HERO SECTION PARALLAX ========== const heroSection = document.getElementById('dynamic-hero'); if (heroSection) { document.addEventListener('mousemove', (e) => { const movementX = (e.clientX / window.innerWidth - 0.5) * 15; const movementY = (e.clientY / window.innerHeight - 0.5) * 15; heroSection.style.backgroundPosition = `calc(center + ${movementX}px) calc(center + ${movementY}px)`; }); } // ========== BACK BUTTON ANIMATION ========== const backButton = document.querySelector('a[href*="products"], a[href*="projects"]'); if (backButton && backButton.textContent.includes('BACK')) { backButton.addEventListener('mouseenter', function() { this.style.transform = 'translateX(-8px)'; }); backButton.addEventListener('mouseleave', function() { this.style.transform = 'translateX(0)'; }); } // ========== SPECS HIGHLIGHT ON SCROLL ========== const observer = new IntersectionObserver((entries) => { entries.forEach(entry => { if (entry.isIntersecting) { entry.target.style.animation = 'slideUp 0.6s ease-out'; } }); }); document.querySelectorAll('h3, h2').forEach(el => { observer.observe(el); }); }); // ========== HELPER FUNCTIONS ========== // Add slide counter display function addSlideCounter() { const wrapper = document.getElementById('slider-wrapper'); if (!wrapper) return; const totalSlides = wrapper.querySelectorAll('img').length; const counter = document.createElement('div'); counter.style.cssText = ` position: absolute; top: 15px; right: 15px; background: rgba(255, 24, 23, 0.8); color: white; padding: 8px 12px; border-radius: 20px; font-size: 12px; z-index: 11; font-weight: bold; backdrop-filter: blur(8px); `; counter.id = 'slide-counter'; counter.textContent = '1 / ' + totalSlides; const container = document.querySelector('.slider-container'); if (container) { container.appendChild(counter); } } // Update slide counter function updateSlideCounter(current) { const counter = document.getElementById('slide-counter'); if (counter) { const wrapper = document.getElementById('slider-wrapper'); const total = wrapper.querySelectorAll('img').length; counter.textContent = (current + 1) + ' / ' + total; } } // Open image in modal function openImageModal(imageSrc) { const modal = document.createElement('div'); modal.style.cssText = ` position: fixed; top: 0; left: 0; width: 100%; height: 100%; background: rgba(0, 0, 0, 0.95); z-index: 1000; display: flex; align-items: center; justify-content: center; animation: fadeIn 0.3s ease-out; `; const img = document.createElement('img'); img.src = imageSrc; img.style.cssText = ` max-width: 90%; max-height: 90%; border-radius: 10px; animation: zoomIn 0.4s cubic-bezier(0.34, 1.56, 0.64, 1); `; const closeBtn = document.createElement('button'); closeBtn.textContent = '✕'; closeBtn.style.cssText = ` position: absolute; top: 20px; right: 30px; background: rgba(255, 24, 23, 0.8); color: white; border: none; width: 40px; height: 40px; border-radius: 50%; font-size: 24px; cursor: pointer; transition: all 0.3s ease; z-index: 1001; `; closeBtn.addEventListener('mouseenter', () => { closeBtn.style.background = 'rgba(255, 24, 23, 1)'; closeBtn.style.transform = 'scale(1.1)'; }); closeBtn.addEventListener('mouseleave', () => { closeBtn.style.background = 'rgba(255, 24, 23, 0.8)'; closeBtn.style.transform = 'scale(1)'; }); closeBtn.addEventListener('click', () => { modal.style.animation = 'fadeOut 0.3s ease-out'; setTimeout(() => modal.remove(), 300); }); modal.addEventListener('click', (e) => { if (e.target === modal) { modal.style.animation = 'fadeOut 0.3s ease-out'; setTimeout(() => modal.remove(), 300); } }); modal.appendChild(img); modal.appendChild(closeBtn); document.body.appendChild(modal); // Press ESC to close const escHandler = (e) => { if (e.key === 'Escape') { modal.style.animation = 'fadeOut 0.3s ease-out'; setTimeout(() => { modal.remove(); document.removeEventListener('keydown', escHandler); }, 300); } }; document.addEventListener('keydown', escHandler); } // Override moveSlide to update counter const originalMoveSlide = window.moveSlide; window.moveSlide = function(direction) { if (originalMoveSlide) { originalMoveSlide(direction); } const wrapper = document.getElementById('slider-wrapper'); if (wrapper) { const currentSlide = Math.round( -wrapper.style.transform.match(/-?\d+/)[0] / 100 ); updateSlideCounter(currentSlide); } }; // ========== SMOOTH SCROLL FOR ANCHORS ========== document.querySelectorAll('a[href^="#"]').forEach(anchor => { anchor.addEventListener('click', function (e) { e.preventDefault(); const target = document.querySelector(this.getAttribute('href')); if (target) { target.scrollIntoView({ behavior: 'smooth', block: 'start' }); } }); }); // ========== ADD GLOW EFFECT TO IMPORTANT TEXT ========== document.addEventListener('DOMContentLoaded', () => { const specs = document.querySelectorAll('td[id^="spec-"]'); specs.forEach(spec => { if (spec.textContent.includes('Available') || spec.textContent.includes('Active') || spec.textContent.includes('12')) { spec.style.color = '#FF1817'; spec.style.fontWeight = 'bold'; } }); }); console.log('[v0] Detail Pages Interactions Loaded');
